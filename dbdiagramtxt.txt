  id int PK
  name varchar
  number int
}

Table Tracks {
  id int PK
  name varchar
}

Table Seasons {
  id int PK
  year int
  team int
  driver int
  engine int
}

Table Races {
  id int PK
  track int
  season int
}

Table DriverResult {
  id int PK
  driver int
  race int
}

Table Teams {
  id int PK
  name varchar
}

Table TeamResult {
  id int PK
  team int PK
  race int
}

Table Engines {
  id int PK
  name varchar
}

Table SeasonDriver {
  id int PK
  driver int
  skill int 
}

Table SeasonTeam {
  id int PK
  team int
  engine int
  chassis int
}

Table SeasonEngine {
  id int PK
  engine int
  team int
  power int
}

Table Qualification {
  id int PK
}
